import {
  View,
  Text,
  StyleSheet,
  TouchableOpacity,
  ScrollView,
} from "react-native";
import { useLocalSearchParams, useRouter } from "expo-router";
import { useEffect, useState } from "react";
import { Ionicons } from "@expo/vector-icons";

export default function CircleProfile() {
  const { id } = useLocalSearchParams();
  const router = useRouter();

  const circleId = Array.isArray(id) ? id[0] : id;

  const [circle, setCircle] = useState<any>(null);
  const [activeTab, setActiveTab] = useState<"history" | "live">("history");

  useEffect(() => {
    if (!circleId) return;

    fetch(`http://localhost:3001/circles/${circleId}`)
      .then((res) => res.json())
      .then((data) => setCircle(data))
      .catch((err) => console.error(err));
  }, [circleId]);

  if (!circle) {
    return (
      <View style={styles.container}>
        <Text style={{ color: "white", marginTop: 100 }}>
          Loading...
        </Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      {/* HEADER */}
      <View style={styles.topHeader}>
        <TouchableOpacity
          style={styles.backRow}
          onPress={() => router.back()}
        >
          <Ionicons name="arrow-back" size={22} color="white" />
          <Text style={styles.backText}>Circle Profile</Text>
        </TouchableOpacity>
      </View>

      <ScrollView contentContainerStyle={{ paddingBottom: 100 }}>
        {/* PROFILE SECTION */}
        <View style={styles.profileSection}>
          <View style={styles.circleIcon}>
            <Ionicons name="people" size={40} color="white" />
          </View>

          <Text style={styles.circleName}>{circle.name}</Text>

          {circle.description ? (
            <Text style={styles.circleDescription}>
              {circle.description}
            </Text>
          ) : null}

          {/* BUTTON ROW */}
          <View style={styles.buttonRow}>
            <TouchableOpacity style={styles.sideButton}>
              <Text style={styles.sideButtonText}>See Members</Text>
            </TouchableOpacity>

            <TouchableOpacity
              style={styles.editButton}
              onPress={() =>
                router.push(
                    {pathname:`/circle-profile/${circleId}/edit`}
                )
              }
            >
              <Text style={styles.editButtonText}>Edit Circle</Text>
            </TouchableOpacity>

            <TouchableOpacity style={styles.sideButton}>
              <Text style={styles.sideButtonText}>Add Friend</Text>
            </TouchableOpacity>
          </View>
        </View>

        {/* TABS */}
        <View style={styles.tabs}>
          <TouchableOpacity
            style={
              activeTab === "history"
                ? styles.activeTab
                : styles.inactiveTab
            }
            onPress={() => setActiveTab("history")}
          >
            <Text style={styles.tabText}>Circle History</Text>
          </TouchableOpacity>

          <TouchableOpacity
            style={
              activeTab === "live"
                ? styles.activeTab
                : styles.inactiveTab
            }
            onPress={() => setActiveTab("live")}
          >
            <Text style={styles.tabText}>Live Bets</Text>
          </TouchableOpacity>
        </View>

        <View style={styles.content}>
          <Text style={styles.historyItem}>
            History coming from DB soon
          </Text>
        </View>
      </ScrollView>
    </View>
  );
}

    flex: 1,
    backgroundColor: "#0B1C2D",
  },
  topHeader: {

    paddingHorizontal: 20,
  },

  },

    color: "white",

    marginLeft: 8,
  },
  profileSection: {
    alignItems: "center",
    marginTop: 30,
  },
  circleIcon: {
    backgroundColor: "#2E5BEA",
    width: 120,
    height: 120,
    borderRadius: 60,

    alignItems: "center",
  },
  circleName: {

    marginTop: 15,
  },
  circleDescription: {
    color: "#8FA3BF",
    marginTop: 8,

    paddingHorizontal: 30,
  },
  buttonRow: {
    flexDirection: "row",
    alignItems: "center",

    marginTop: 20,
  },
  editButton: {
    backgroundColor: "#2E5BEA",
    paddingVertical: 12,
    paddingHorizontal: 22,
    borderRadius: 12,
    marginHorizontal: 10,
  },
  editButtonText: {

  },
